import pickle
import os

def add():
    F = open("bid.dat","wb")
    list1 = []
    while True:
        print("*"*30,"Entry of a record","*"*30)
        bid = int(input("Enter B.Id: "))
        arrival = str(input("Enter arrival: "))
        destination = str(input("Enter destination: "))
        doj = str(input("Enter doj: "))
        fare = float(input("Enter fare: "))
        list = [bid,arrival,destination,doj,fare]
        list1.append(list)
        print("*"*32,"Entry Inserted","*"*31)
        print()
        cont = input("Would you like to add more records? ")
        print()
        if cont == 'yes' or cont == "Yes":
            continue
        else:
            break
    pickle.dump(list1,F)
    F.close()

def delete():
    F = open("bid.dat","rb")
    F1 = open("temp.dat","wb")
    data = pickle.load(F)
    print("*"*30,"Deletion of a record","*"*30)
    bid2 = int(input("Enter B.ID whose record you'd like to delete: "))
    for i in data:
        if bid2 != i[0]:
            L = [i[0],i[1],i[2],i[3],i[4]]
            pickle.dump(L,F1)
    F.close()
    F1.close()
    print("*"*23,"Record successfully deleted","*"*23)
    print()
    os.remove("bid.dat")
    os.rename("temp.dat","bid.dat")

def display():
    F = open("bid.dat","rb")
    count = 1
    try:
        data = pickle.load(F)
        for i in data:
            print("DATA:",count)
            print()
            print("B.Id:",i[0])
            print("Arrival:",i[1])
            print("Destintion:",i[2])
            print("DOJ:",i[3])
            print("Fare(in rupees):",i[4])
            count+=1
            print()
            print()
    except:
        print("End of File.")
    F.close()
